65 // Storage local resource provider initially picks a random amount of time
66 // between `[0, b]`, where `b = DEFAULT_CSI_RETRY_BACKOFF_FACTOR`, to retry CSI
67 // calls related to `CREATE_DISK` or `DESTROY_DISK` operations. Subsequent
68 // retries are exponentially backed off based on this interval (e.g., 2nd retry
69 // uses a random value between `[0, b * 2^1]`, 3rd retry between `[0, b * 2^2]`,
70 // etc) up to a maximum of `DEFAULT_CSI_RETRY_INTERVAL_MAX`.
71 //
72 // TODO(chhsiao): Make the retry parameters configurable.
73 constexpr Duration DEFAULT_CSI_RETRY_BACKOFF_FACTOR = Seconds(10);
74 constexpr Duration DEFAULT_CSI_RETRY_INTERVAL_MAX = Minutes(10);

99  // Wrapper functions to make CSI calls and update RPC metrics. Made public for
100  // testing purpose.
101  //
102  // The call is made asynchronously and thus no guarantee is provided on the
103  // order in which calls are sent. Callers need to either ensure to not have
104  // multiple conflicting calls in flight, or treat results idempotently.
105  //
106  // NOTE: We currently ensure this by 1) resource locking to forbid concurrent
107  // calls on the same volume, and 2) no profile update while there are ongoing
108  // `CREATE_DISK` or `DESTROY_DISK` operations.
109  //
110  // NOTE: Since this function uses `getService` to obtain the latest service
111  // future, which depends on probe results, it is disabled for making probe
112  // calls; `_call` should be used directly instead.
113  template <
114  csi::v0::RPC rpc,
115  typename std::enable_if<rpc != csi::v0::PROBE, int>::type = 0>
116  process::Future<csi::v0::Response<rpc>> call(
117  const csi::Service& service,
118  const csi::v0::Request<rpc>& request,
