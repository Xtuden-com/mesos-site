67 // Write out the given protobuf to the specified file descriptor by
68 // first writing out the length of the protobuf followed by the
69 // contents.
70 // NOTE: On error, this may have written partial data to the file.
71 inline Try<Nothing> write(int_fd fd, const google::protobuf::Message& message)
72 {
73  if (!message.IsInitialized()) {
74  return Error(message.InitializationErrorString() +
75  " is required but not initialized");
76  }
77 
78  // First write the size of the protobuf.
79  uint32_t size = message.ByteSize();
80  std::string bytes((char*) &size, sizeof(size));
81 
82  Try<Nothing> result = os::write(fd, bytes);
83  if (result.isError()) {
84  return Error("Failed to write size: " + result.error());
85  }
86 
87 #ifdef __WINDOWS__
88  // NOTE: On Windows, we need to explicitly allocate a CRT file
89  // descriptor because the Protobuf library requires it. Because
90  // users of `protobuf::write` are likely to call `os::close` on the
91  // `fd` we were given, we need to duplicate it before allocating the
92  // CRT fd. This is because once the CRT fd is allocated, it must be
93  // closed with `_close` instead of `os::close`. Since we need to
94  // call `_close` here, we duplicate the fd to prevent the users call
95  // of `os::close` from closing twice.
96  Try<int_fd> dup = os::dup(fd);
97  if (dup.isError()) {
98  return Error("Failed to duplicate handle: " + dup.error());
99  }
100 
101  int crt = dup->crt();
102 
103  if (!message.SerializeToFileDescriptor(crt)) {
104  ::_close(crt);
105  return Error("Failed to write/serialize message");
106  }
107  ::_close(crt);
108 #else
109  if (!message.SerializeToFileDescriptor(fd)) {
110  return Error("Failed to write/serialize message");
111  }
112 #endif
113 
114  return Nothing();
115 }

117 // Write out the given sequence of protobuf messages to the
118 // specified file descriptor by repeatedly invoking write
119 // on each of the messages.
120 // NOTE: On error, this may have written partial data to the file.
121 template <typename T>
122 Try<Nothing> write(
123  int_fd fd, const google::protobuf::RepeatedPtrField<T>& messages)
124 {
125  foreach (const T& message, messages) {
126  Try<Nothing> result = write(fd, message);
127  if (result.isError()) {
128  return Error(result.error());
129  }
130  }
131 
132  return Nothing();
133 }

136 // A wrapper function for the above `write()` with opening and closing the file.
137 // If `sync` is set to true, an `fsync()` will be called before `close()`.
138 template <typename T>
139 Try<Nothing> write(const std::string& path, const T& t, bool sync = false)
140 {
141  Try<int_fd> fd = os::open(
142  path,
143  O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
144  S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
145 
146  if (fd.isError()) {
147  return Error("Failed to open file '" + path + "': " + fd.error());
148  }
149 
150  Try<Nothing> write = ::protobuf::write(fd.get(), t);
151 
152  if (sync && write.isSome()) {
153  // We call `fsync()` before closing the file instead of opening it with the
154  // `O_SYNC` flag for better performance. See:
155  // http://lkml.iu.edu/hypermail/linux/kernel/0105.3/0353.html
156  write = os::fsync(fd.get());
157  }
158 
159  Try<Nothing> close = os::close(fd.get());
160 
161  // We propagate `close` failures if `write` on the file was successful.
162  if (write.isSome() && close.isError()) {
163  return Error(
164  "Failed to close '" + stringify(fd.get()) + "':" + close.error());
165  }
166 
167  return write;
168 }

173 inline Try<Nothing> append(

207 template <typename T>
208 Try<T> deserialize(const std::string& value)
209 {
210  T t;
211  (void) static_cast<google::protobuf::Message*>(&t);
212 
213  // Verify that the size of `value` fits into `ArrayInputStream`'s
214  // constructor. The maximum size of a proto2 message is 64 MB, so it is
215  // unlikely that we will hit this limit, but since an arbitrary string can be
216  // passed in, we include this check to be safe.
217  CHECK_LE(value.size(), static_cast<size_t>(std::numeric_limits<int>::max()));
218  google::protobuf::io::ArrayInputStream stream(
219  value.data(),
220  static_cast<int>(value.size()));
221  if (!t.ParseFromZeroCopyStream(&stream)) {
222  return Error("Failed to deserialize " + t.GetDescriptor()->full_name());
223  }
224  return t;
225 }

243 // Reads a single message of type T from the file by first reading the
244 // "size" followed by the contents (as written by 'write' above).
245 // NOTE: This struct is used by the public 'read' function.
246 // See comments there for the reason why we need this.
247 template <typename T>
248 struct Read
249 {
250  Result<T> operator()(int_fd fd, bool ignorePartial, bool undoFailed)
251  {
252  off_t offset = 0;
253 
254  if (undoFailed) {
255  // Save the offset so we can re-adjust if something goes wrong.
256  Try<off_t> lseek = os::lseek(fd, offset, SEEK_CUR);
257  if (lseek.isError()) {
258  return Error(lseek.error());
259  }
260 
261  offset = lseek.get();
262  }
263 
264  uint32_t size;
265  Result<std::string> result = os::read(fd, sizeof(size));
266 
267  if (result.isError()) {
268  if (undoFailed) {
269  os::lseek(fd, offset, SEEK_SET);
270  }
271  return Error("Failed to read size: " + result.error());
272  } else if (result.isNone()) {
273  return None(); // No more protobufs to read.
274  } else if (result->size() < sizeof(size)) {
275  // Hit EOF unexpectedly.
276  if (undoFailed) {
277  // Restore the offset to before the size read.
278  os::lseek(fd, offset, SEEK_SET);
279  }
280  if (ignorePartial) {
281  return None();
282  }
283  return Error(
284  "Failed to read size: hit EOF unexpectedly, possible corruption");
285  }
286 
287  // Parse the size from the bytes.
288  memcpy((void*)&size, (void*)result->data(), sizeof(size));
289 
290  // NOTE: Instead of specifically checking for corruption in 'size',
291  // we simply try to read 'size' bytes. If we hit EOF early, it is an
292  // indication of corruption.
293  result = os::read(fd, size);
294 
295  if (result.isError()) {
296  if (undoFailed) {
297  // Restore the offset to before the size read.
298  os::lseek(fd, offset, SEEK_SET);
299  }
300  return Error("Failed to read message: " + result.error());
301  } else if (result.isNone() || result->size() < size) {
302  // Hit EOF unexpectedly.
303  if (undoFailed) {
304  // Restore the offset to before the size read.
305  os::lseek(fd, offset, SEEK_SET);
306  }
307  if (ignorePartial) {
308  return None();
309  }
310  return Error("Failed to read message of size " + stringify(size) +
311  " bytes: hit EOF unexpectedly, possible corruption");
312  }
313 
314  // Parse the protobuf from the string.
315  // NOTE: We need to capture a const reference to the data because it
316  // must outlive the creation of ArrayInputStream.
317  const std::string& data = result.get();
318 
319  // Verify that the size of `data` fits into `ArrayInputStream`'s
320  // constructor. The maximum size of a proto2 message is 64 MB, so it is
321  // unlikely that we will hit this limit, but since an arbitrary string can
322  // be passed in, we include this check to be safe.
323  CHECK_LE(data.size(), static_cast<size_t>(std::numeric_limits<int>::max()));
324  T message;
325  google::protobuf::io::ArrayInputStream stream(
326  data.data(),
327  static_cast<int>(data.size()));
328 
329  if (!message.ParseFromZeroCopyStream(&stream)) {
330  if (undoFailed) {
331  // Restore the offset to before the size read.
332  os::lseek(fd, offset, SEEK_SET);
333  }
334  return Error("Failed to deserialize message");
335  }
336 
337  return message;
338  }
339 };

342 // Partial specialization for RepeatedPtrField<T> to read a sequence
343 // of protobuf messages from a given fd by repeatedly invoking
344 // Read<T> until None is reached, which we treat as EOF.
345 // NOTE: This struct is used by the public 'read' function.
346 // See comments there for the reason why we need this.
347 template <typename T>
348 struct Read<google::protobuf::RepeatedPtrField<T>>
349 {
350  Result<google::protobuf::RepeatedPtrField<T>> operator()(
351  int_fd fd, bool ignorePartial, bool undoFailed)
352  {
353  google::protobuf::RepeatedPtrField<T> result;
354  for (;;) {
355  Result<T> message = Read<T>()(fd, ignorePartial, undoFailed);
356  if (message.isError()) {
357  return Error(message.error());
358  } else if (message.isNone()) {
359  break;
360  } else {
361  result.Add()->CopyFrom(message.get());
362  }
363  }
364  return result;
365  }
366 };

371 // Reads the protobuf message(s) from a given fd based on the format
372 // written by write() above. We use partial specialization of
373 // - internal::Read<T> vs
374 // - internal::Read<google::protobuf::RepeatedPtrField<T>>
375 // in order to determine whether T is a single protobuf message or
376 // a sequence of messages.
377 // If 'ignorePartial' is true, None() is returned when we unexpectedly
378 // hit EOF while reading the protobuf (e.g., partial write).
379 // If 'undoFailed' is true, failed read attempts will restore the file
380 // read/write file offset towards the initial callup position.
381 template <typename T>
382 Result<T> read(int_fd fd, bool ignorePartial = false, bool undoFailed = false)
383 {
384  return internal::Read<T>()(fd, ignorePartial, undoFailed);
385 }

388 // A wrapper function that wraps the above read() with open and
389 // closing the file.
390 template <typename T>
391 Result<T> read(const std::string& path)
392 {
393  Try<int_fd> fd = os::open(
394  path,
395  O_RDONLY | O_CLOEXEC,
396  S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
397 
398  if (fd.isError()) {
399  return Error("Failed to open file '" + path + "': " + fd.error());
400  }
401 
402  Result<T> result = read<T>(fd.get());
403 
404  // NOTE: We ignore the return value of close(). This is because
405  // users calling this function are interested in the return value of
406  // read(). Also an unsuccessful close() doesn't affect the read.
407  os::close(fd.get());
408 
409  return result;
410 }

421 struct Parser : boost::static_visitor<Try<Nothing>>
422 {
423  Parser(google::protobuf::Message* _message,
424  const google::protobuf::FieldDescriptor* _field)
425  : message(_message),
426  reflection(message->GetReflection()),
427  field(_field) {}
428 
429  Try<Nothing> operator()(const JSON::Object& object) const
430  {
431  switch (field->type()) {
432  case google::protobuf::FieldDescriptor::TYPE_MESSAGE:
433  if (field->is_map()) {
434  foreachpair (
435  const std::string& name,
436  const JSON::Value& value,
437  object.values) {
438  google::protobuf::Message* entry =
439  reflection->AddMessage(message, field);
440 
441  // A map is equivalent to:
442  //
443  // message MapFieldEntry {
444  // optional key_type key = 1;
445  // optional value_type value = 2;
446  // }
447  //
448  // repeated MapFieldEntry map_field = N;
449  //
450  // See the link below for details:
451  // https://developers.google.com/protocol-buffers/docs/proto#maps
452  const google::protobuf::FieldDescriptor* key_field =
453  entry->GetDescriptor()->FindFieldByNumber(1);
454 
455  JSON::Value key(name);
456 
457  Try<Nothing> apply =
458  boost::apply_visitor(Parser(entry, key_field), key);
459 
460  if (apply.isError()) {
461  return Error(apply.error());
462  }
463 
464  const google::protobuf::FieldDescriptor* value_field =
465  entry->GetDescriptor()->FindFieldByNumber(2);
466 
467  apply = boost::apply_visitor(Parser(entry, value_field), value);
468  if (apply.isError()) {
469  return Error(apply.error());
470  }
471  }
472  } else if (field->is_repeated()) {
473  // TODO(gilbert): We currently push up the nested error
474  // messages without wrapping the error message (due to
475  // the recursive nature of parse). We should pass along
476  // variable information in order to construct a helpful
477  // error message, e.g. "Failed to parse field 'a.b.c': ...".
478  return parse(reflection->AddMessage(message, field), object);
479  } else {
480  return parse(reflection->MutableMessage(message, field), object);
481  }
482  break;
483  default:
484  return Error("Not expecting a JSON object for field '" +
485  field->name() + "'");
486  }
487  return Nothing();
488  }
489 
490  Try<Nothing> operator()(const JSON::String& string) const
491  {
492  switch (field->type()) {
493  case google::protobuf::FieldDescriptor::TYPE_STRING:
494  if (field->is_repeated()) {
495  reflection->AddString(message, field, string.value);
496  } else {
497  reflection->SetString(message, field, string.value);
498  }
499  break;
500  case google::protobuf::FieldDescriptor::TYPE_BYTES: {
501  Try<std::string> decode = base64::decode(string.value);
502  if (decode.isError()) {
503  return Error("Failed to base64 decode bytes field"
504  " '" + field->name() + "': " + decode.error());
505  }
506 
507  if (field->is_repeated()) {
508  reflection->AddString(message, field, decode.get());
509  } else {
510  reflection->SetString(message, field, decode.get());
511  }
512  break;
513  }
514  case google::protobuf::FieldDescriptor::TYPE_ENUM: {
515  const google::protobuf::EnumValueDescriptor* descriptor =
516  field->enum_type()->FindValueByName(string.value);
517 
518  if (descriptor == nullptr) {
519  if (field->is_required()) {
520  return Error("Failed to find enum for '" + string.value + "'");
521  }
522 
523  // Unrecognized enum value will be discarded if this is not a
524  // required enum field, which makes the field's `has..` accessor
525  // return false and its getter return the first value listed in
526  // the enum definition, or the default value if one is specified.
527  //
528  // This is the deserialization behavior of proto2, see the link
529  // below for details:
530  // https://developers.google.com/protocol-buffers/docs/proto#updating
531  break;
532  }
533 
534  if (field->is_repeated()) {
535  reflection->AddEnum(message, field, descriptor);
536  } else {
537  reflection->SetEnum(message, field, descriptor);
538  }
539  break;
540  }
541  case google::protobuf::FieldDescriptor::TYPE_DOUBLE:
542  case google::protobuf::FieldDescriptor::TYPE_FLOAT:
543  case google::protobuf::FieldDescriptor::TYPE_INT64:
544  case google::protobuf::FieldDescriptor::TYPE_SINT64:
545  case google::protobuf::FieldDescriptor::TYPE_SFIXED64:
546  case google::protobuf::FieldDescriptor::TYPE_UINT64:
547  case google::protobuf::FieldDescriptor::TYPE_FIXED64:
548  case google::protobuf::FieldDescriptor::TYPE_INT32:
549  case google::protobuf::FieldDescriptor::TYPE_SINT32:
550  case google::protobuf::FieldDescriptor::TYPE_SFIXED32:
551  case google::protobuf::FieldDescriptor::TYPE_UINT32:
552  case google::protobuf::FieldDescriptor::TYPE_FIXED32: {
553  Try<JSON::Number> number = JSON::parse<JSON::Number>(string.value);
554  if (number.isError()) {
555  return Error(
556  "Failed to parse '" + string.value + "' as a JSON number "
557  "for field '" + field->name() + "': " + number.error());
558  }
559 
560  return operator()(number.get());
561  }
562  case google::protobuf::FieldDescriptor::TYPE_BOOL: {
563  Try<JSON::Boolean> boolean = JSON::parse<JSON::Boolean>(string.value);
564  if (boolean.isError()) {
565  return Error(
566  "Failed to parse '" + string.value + "' as a JSON boolean "
567  "for field '" + field->name() + "': " + boolean.error());
568  }
569 
570  return operator()(boolean.get());
571  }
572  default:
573  return Error("Not expecting a JSON string for field '" +
574  field->name() + "'");
575  }
576  return Nothing();
577  }

756  static_assert(std::is_convertible<T*, google::protobuf::Message*>::value,
757  "T must be a protobuf message");

760  if (array == nullptr) {

772  }

774  collection.Add()->CopyFrom(message.get());
775  }

778  }

809 // `json` function for protobuf messages. Refer to `jsonify.hpp` for details.
810 // TODO(mpark): This currently uses the default value for optional fields
811 // that are not deprecated, but we may want to revisit this decision.
812 inline void json(ObjectWriter* writer, const Protobuf& protobuf)
813 {
814  using google::protobuf::FieldDescriptor;
815 
816  const google::protobuf::Message& message = protobuf;
817 
818  const google::protobuf::Descriptor* descriptor = message.GetDescriptor();
819  const google::protobuf::Reflection* reflection = message.GetReflection();
820 
821  // We first look through all the possible fields to determine both the set
822  // fields __and__ the optional fields with a default that are not set.
823  // `Reflection::ListFields()` alone will only include set fields and
824  // is therefore insufficient.
825  int fieldCount = descriptor->field_count();
826  std::vector<const FieldDescriptor*> fields;
827  fields.reserve(fieldCount);
828  for (int i = 0; i < fieldCount; ++i) {
829  const FieldDescriptor* field = descriptor->field(i);
830  if (field->is_repeated()) {
831  if (reflection->FieldSize(message, field) > 0) {
832  // Has repeated field with members, output as JSON.
833  fields.push_back(field);
834  }
835  } else if (
836  reflection->HasField(message, field) ||
837  (field->has_default_value() && !field->options().deprecated())) {
838  // Field is set or has default, output as JSON.
839  fields.push_back(field);
840  }
841  }
842 
843  foreach (const FieldDescriptor* field, fields) {
844  if (field->is_repeated()) {
845  writer->field(
846  field->name(),
847  [&field, &reflection, &message](JSON::ArrayWriter* writer) {
848  int fieldSize = reflection->FieldSize(message, field);
849  for (int i = 0; i < fieldSize; ++i) {
850  switch (field->cpp_type()) {
851  case FieldDescriptor::CPPTYPE_BOOL:
852  writer->element(
853  reflection->GetRepeatedBool(message, field, i));
854  break;
855  case FieldDescriptor::CPPTYPE_INT32:
856  writer->element(
857  reflection->GetRepeatedInt32(message, field, i));
858  break;
859  case FieldDescriptor::CPPTYPE_INT64:
860  writer->element(

862  break;
863  case FieldDescriptor::CPPTYPE_UINT32:
864  writer->element(
865  reflection->GetRepeatedUInt32(message, field, i));
866  break;
867  case FieldDescriptor::CPPTYPE_UINT64:
868  writer->element(
869  reflection->GetRepeatedUInt64(message, field, i));
870  break;
871  case FieldDescriptor::CPPTYPE_FLOAT:
872  writer->element(
873  reflection->GetRepeatedFloat(message, field, i));
874  break;
875  case FieldDescriptor::CPPTYPE_DOUBLE:
876  writer->element(
877  reflection->GetRepeatedDouble(message, field, i));
878  break;
879  case FieldDescriptor::CPPTYPE_MESSAGE:
880  writer->element(Protobuf(
881  reflection->GetRepeatedMessage(message, field, i)));
882  break;
883  case FieldDescriptor::CPPTYPE_ENUM:
884  writer->element(
885  reflection->GetRepeatedEnum(message, field, i)->name());
886  break;
887  case FieldDescriptor::CPPTYPE_STRING:
888  const std::string& s = reflection->GetRepeatedStringReference(
889  message, field, i, nullptr);
890  if (field->type() == FieldDescriptor::TYPE_BYTES) {
891  writer->element(base64::encode(s));
892  } else {
893  writer->element(s);
894  }
895  break;
896  }
897  }
898  });
899  } else {
900  switch (field->cpp_type()) {
901  case FieldDescriptor::CPPTYPE_BOOL:
902  writer->field(field->name(), reflection->GetBool(message, field));
903  break;
904  case FieldDescriptor::CPPTYPE_INT32:
905  writer->field(field->name(), reflection->GetInt32(message, field));
906  break;
907  case FieldDescriptor::CPPTYPE_INT64:
908  writer->field(field->name(), reflection->GetInt64(message, field));
909  break;
910  case FieldDescriptor::CPPTYPE_UINT32:
911  writer->field(field->name(), reflection->GetUInt32(message, field));
912  break;
913  case FieldDescriptor::CPPTYPE_UINT64:
914  writer->field(field->name(), reflection->GetUInt64(message, field));
915  break;
916  case FieldDescriptor::CPPTYPE_FLOAT:
917  writer->field(field->name(), reflection->GetFloat(message, field));
918  break;
919  case FieldDescriptor::CPPTYPE_DOUBLE:
920  writer->field(field->name(), reflection->GetDouble(message, field));
921  break;
922  case FieldDescriptor::CPPTYPE_MESSAGE:
923  writer->field(
924  field->name(), Protobuf(reflection->GetMessage(message, field)));
925  break;
926  case FieldDescriptor::CPPTYPE_ENUM:
927  writer->field(
928  field->name(), reflection->GetEnum(message, field)->name());
929  break;
930  case FieldDescriptor::CPPTYPE_STRING:
931  const std::string& s = reflection->GetStringReference(
932  message, field, nullptr);
933  if (field->type() == FieldDescriptor::TYPE_BYTES) {
934  writer->field(field->name(), base64::encode(s));

946 // that are not deprecated, but we may want to revisit this decision.
947 inline Object protobuf(const google::protobuf::Message& message)
948 {
949  Object object;
950 
951  const google::protobuf::Descriptor* descriptor = message.GetDescriptor();
952  const google::protobuf::Reflection* reflection = message.GetReflection();
953 
954  auto value_for_field = [](
955  const google::protobuf::Message& message,
956  const google::protobuf::FieldDescriptor* field) -> JSON::Value {
957  const google::protobuf::Reflection* reflection = message.GetReflection();
958 
959  switch (field->type()) {
960  case google::protobuf::FieldDescriptor::TYPE_DOUBLE:
961  return JSON::Number(reflection->GetDouble(message, field));
962  case google::protobuf::FieldDescriptor::TYPE_FLOAT:
963  return JSON::Number(reflection->GetFloat(message, field));
964  case google::protobuf::FieldDescriptor::TYPE_INT64:
965  case google::protobuf::FieldDescriptor::TYPE_SINT64:
966  case google::protobuf::FieldDescriptor::TYPE_SFIXED64:
967  return JSON::Number(reflection->GetInt64(message, field));
968  case google::protobuf::FieldDescriptor::TYPE_UINT64:
969  case google::protobuf::FieldDescriptor::TYPE_FIXED64:
970  return JSON::Number(reflection->GetUInt64(message, field));
971  case google::protobuf::FieldDescriptor::TYPE_INT32:
972  case google::protobuf::FieldDescriptor::TYPE_SINT32:
973  case google::protobuf::FieldDescriptor::TYPE_SFIXED32:
974  return JSON::Number(reflection->GetInt32(message, field));
975  case google::protobuf::FieldDescriptor::TYPE_UINT32:
976  case google::protobuf::FieldDescriptor::TYPE_FIXED32:
977  return JSON::Number(reflection->GetUInt32(message, field));
978  case google::protobuf::FieldDescriptor::TYPE_BOOL:
979  if (reflection->GetBool(message, field)) {
980  return JSON::Boolean(true);
981  } else {
982  return JSON::Boolean(false);
983  }
984  break;
985  case google::protobuf::FieldDescriptor::TYPE_STRING:
986  return JSON::String(reflection->GetString(message, field));
987  case google::protobuf::FieldDescriptor::TYPE_BYTES:
988  return JSON::String(
989  base64::encode(reflection->GetString(message, field)));
990  case google::protobuf::FieldDescriptor::TYPE_MESSAGE:
991  return protobuf(reflection->GetMessage(message, field));
992  case google::protobuf::FieldDescriptor::TYPE_ENUM:
993  return JSON::String(reflection->GetEnum(message, field)->name());
994  case google::protobuf::FieldDescriptor::TYPE_GROUP:
995  // Deprecated! We abort here instead of using a Try as return value,
996  // because we expect this code path to never be taken.
997  ABORT("Unhandled protobuf field type: " +
998  stringify(field->type()));
999  }
1000 
1001  UNREACHABLE();
1002  };
1003 
1004  // We first look through all the possible fields to determine both
1005  // the set fields _and_ the optional fields with a default that
1006  // are not set. Reflection::ListFields() alone will only include
1007  // set fields and is therefore insufficient.
1008  std::vector<const google::protobuf::FieldDescriptor*> fields;
1009  fields.reserve(descriptor->field_count());
1010  for (int i = 0; i < descriptor->field_count(); i++) {
1011  const google::protobuf::FieldDescriptor* field = descriptor->field(i);
1012  if (field->is_repeated()) {
1013  if (reflection->FieldSize(message, descriptor->field(i)) > 0) {
1014  // Has repeated field with members, output as JSON.
1015  fields.push_back(field);
1016  }
1017  } else if (
1018  reflection->HasField(message, field) ||
1019  (field->has_default_value() && !field->options().deprecated())) {
1020  // Field is set or has default, output as JSON.
1021  fields.push_back(field);
1022  }
1023  }
1024 
1025  foreach (const google::protobuf::FieldDescriptor* field, fields) {
1026  if (field->is_map()) {
1027  JSON::Object map;
1028 
1029  int fieldSize = reflection->FieldSize(message, field);
1030  for (int i = 0; i < fieldSize; ++i) {
1031  const google::protobuf::Message& entry =
1032  reflection->GetRepeatedMessage(message, field, i);
1033 
1034  // A map is equivalent to:
1035  //
1036  // message MapFieldEntry {
1037  // optional key_type key = 1;
1038  // optional value_type value = 2;
1039  // }
1040  //
1041  // repeated MapFieldEntry map_field = N;
1042  //
1043  // See the link below for details:
1044  // https://developers.google.com/protocol-buffers/docs/proto#maps
1045  const google::protobuf::FieldDescriptor* key_field =
1046  entry.GetDescriptor()->FindFieldByNumber(1);
1047 
1048  const google::protobuf::FieldDescriptor* value_field =
1049  entry.GetDescriptor()->FindFieldByNumber(2);
1050 
1051  JSON::Value key = value_for_field(entry, key_field);
1052 
1053  std::string name;
1054  if (key.is<JSON::String>()) {
1055  name = key.as<JSON::String>().value;
1056  } else {
1057  name = jsonify(key);
1058  }
1059 
1060  map.values[name] = value_for_field(entry, value_field);
1061  }
1062  object.values[field->name()] = map;
1063  } else if (field->is_repeated()) {
1064  JSON::Array array;
1065  int fieldSize = reflection->FieldSize(message, field);
1066  array.values.reserve(fieldSize);
1067  for (int i = 0; i < fieldSize; ++i) {
1068  switch (field->type()) {
1069  case google::protobuf::FieldDescriptor::TYPE_DOUBLE:
1070  array.values.push_back(JSON::Number(
1071  reflection->GetRepeatedDouble(message, field, i)));
1072  break;
1073  case google::protobuf::FieldDescriptor::TYPE_FLOAT:
1074  array.values.push_back(JSON::Number(
1075  reflection->GetRepeatedFloat(message, field, i)));
1076  break;
1077  case google::protobuf::FieldDescriptor::TYPE_INT64:
1078  case google::protobuf::FieldDescriptor::TYPE_SINT64:
1079  case google::protobuf::FieldDescriptor::TYPE_SFIXED64:
1080  array.values.push_back(JSON::Number(
1081  reflection->GetRepeatedInt64(message, field, i)));
1082  break;
1083  case google::protobuf::FieldDescriptor::TYPE_UINT64:
1084  case google::protobuf::FieldDescriptor::TYPE_FIXED64:
1085  array.values.push_back(JSON::Number(
1086  reflection->GetRepeatedUInt64(message, field, i)));
1087  break;
1088  case google::protobuf::FieldDescriptor::TYPE_INT32:
1089  case google::protobuf::FieldDescriptor::TYPE_SINT32:
1090  case google::protobuf::FieldDescriptor::TYPE_SFIXED32:
1091  array.values.push_back(JSON::Number(
1092  reflection->GetRepeatedInt32(message, field, i)));
1093  break;
1094  case google::protobuf::FieldDescriptor::TYPE_UINT32:
1095  case google::protobuf::FieldDescriptor::TYPE_FIXED32:
1096  array.values.push_back(JSON::Number(
1097  reflection->GetRepeatedUInt32(message, field, i)));
1098  break;
1099  case google::protobuf::FieldDescriptor::TYPE_BOOL:
1100  if (reflection->GetRepeatedBool(message, field, i)) {
1101  array.values.push_back(JSON::Boolean(true));
1102  } else {
1103  array.values.push_back(JSON::Boolean(false));
1104  }
1105  break;
1106  case google::protobuf::FieldDescriptor::TYPE_STRING:
1107  array.values.push_back(JSON::String(
1108  reflection->GetRepeatedString(message, field, i)));
1109  break;
1110  case google::protobuf::FieldDescriptor::TYPE_BYTES:
1111  array.values.push_back(JSON::String(base64::encode(
1112  reflection->GetRepeatedString(message, field, i))));
1113  break;
1114  case google::protobuf::FieldDescriptor::TYPE_MESSAGE:
1115  array.values.push_back(protobuf(
1116  reflection->GetRepeatedMessage(message, field, i)));
1117  break;
1118  case google::protobuf::FieldDescriptor::TYPE_ENUM:
1119  array.values.push_back(JSON::String(
1120  reflection->GetRepeatedEnum(message, field, i)->name()));
1121  break;
1122  case google::protobuf::FieldDescriptor::TYPE_GROUP:
1123  // Deprecated! We abort here instead of using a Try as return value,
1124  // because we expect this code path to never be taken.
1125  ABORT("Unhandled protobuf field type: " +
1126  stringify(field->type()));
1127  }
1128  }
1129  object.values[field->name()] = array;
1130  } else {
1131  object.values[field->name()] = value_for_field(message, field);
1132  }
1133  }
1134 
1135  return object;
