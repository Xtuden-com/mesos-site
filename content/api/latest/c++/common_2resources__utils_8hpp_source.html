82 // Resource format options to be used with the `convertResourceFormat` function.
83 //
84 // The preconditions of the options are asymmetric, centered around the
85 // "post-reservation-refinement" format. This is mainly due to the fact that
86 // "post-reservation-refinement" format is our canonical representation.
87 // The transformations are generally applied to any of the 3 formats to be
88 // converted to the canonical format, then later converted back as necessary.
89 //
90 // See 'Resource Format' section in `mesos.proto` for more details.
91 enum ResourceFormat
92 {
93  // "post-reservation-refinement" -> "pre-reservation-refinement"
94  //
95  // The `Resource` objects must be in the "post-reservation-refinement" format,
96  // and must not have refined reservations.
97  //
98  // All resources end up with the `Resource.role` and `Resource.reservation`
99  // fields set, and the `Resource.reservations` field unset.
100  //
101  // We convert the resources to the "pre-reservation-refinement" format to
102  // checkpoint resources for example. This enables downgrading to an agent
103  // without a RESERVATION_REFINEMENT caapbility, since the resources will
104  // be checkpointed in a format that the downgraded agent can recover from.
105  PRE_RESERVATION_REFINEMENT,
106 
107  // "pre-reservation-refinement" -> "post-reservation-refinement"
108  // "post-reservation-refinement" -> "post-reservation-refinement"
109  // "endpoint" -> "post-reservation-refinement"
110  //
111  // The `Resource` objects can be in any of the valid resource formats:
112  // "pre-reservation-refinement", "post-reservation-refinement", "endpoint".
113  //
114  // All resources end up with the `Resource.reservations` field set,
115  // and the `Resource.role` and `Resource.reservation` fields unset.
116  //
117  // If the `Resource` objects are already in the "post-reservation-refinement"
118  // format, this is a no-op.
119  //
120  // We convert the resources to the "post-reservation-refinement" format,
121  // for example, when a master receives a message from an agent without
122  // the RESERVATION_REFINEMENT capability. This allows a component
123  // (e.g. master) code to deal with a canonical format to simplify the code.
124  POST_RESERVATION_REFINEMENT,
125 
126  // "post-reservation-refinement" -> "endpoint"
127  //
128  // This is a special case for endpoints, which injects
129  // the "pre-reservation-refinement" format.
130  //
131  // The `Resource` objects must be in the "post-reservation-refinement" format.
132  //
133  // All resources continue to have the `Resource.reservations` field set.
134  // The `Resource` objects without refined reservations end up with the
135  // `Resource.role` and `Resource.reservation` fields set, and the objects
136  // with refined reservations have them unset.
137  //
138  // We inject the resources with the "pre-reservation-refinement" format to
139  // enable backward compatibility with external tooling. If the master has been
140  // upgraded to a version that supports reservation refinement but no refined
141  // reservations have been made, the endpoints will return the data in both new
142  // and old formats to maximize backward compatibility. However, once
143  // a reservation refinement is made to a resource, that resource is only
144  // returned in the new format.
145  ENDPOINT
146 };

193 // Convert any resources contained in the given message(s)
194 // to the "pre-reservation-refinement" format, if possible.
195 //
196 // These functions do not provide "all-or-nothing" semantics.
197 // The resources are downgraded to "pre-" format until either
198 // (1) there are no more resources, or
199 // (2) a non-downgradable resource is encountered.
200 //
201 // For (1), `Nothing` is returned.
202 // For (2), `Error` is returned, and the rest of the resources are untouched.
203 //

205 // be downgraded to a version that does not support reservation refinement.
