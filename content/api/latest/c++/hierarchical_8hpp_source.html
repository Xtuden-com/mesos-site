232 private:
233  void updateAvailable() {
234  // In order to subtract from the total,
235  // we strip the allocation information.
236  Resources allocated_ = allocated;
237  allocated_.unallocate();
238 
239  // Calling `nonShared()` currently copies the underlying resources
240  // and is therefore rather expensive. We avoid it in the common
241  // case that there are no shared resources.
242  //
243  // TODO(mzhu): Ideally there would be a single logical path here.
244  // One solution is to have `Resources` be copy-on-write such that
245  // `nonShared()` performs no copying and instead points to a
246  // subset of the original `Resource` objects.
247  if (shared.empty()) {
248  available = total - allocated_;
249  } else {
250  // Since shared resources are offerable even when they are in use, we
251  // always include them as part of available resources.
252  available = (total.nonShared() - allocated_.nonShared()) + shared;
253  }
254  }
255 
256  // Total amount of regular *and* oversubscribed resources.
257  Resources total;
258 
259  // Regular *and* oversubscribed resources that are allocated.
260  //
261  // NOTE: We maintain multiple copies of each shared resource allocated
262  // to a slave, where the number of copies represents the number of times
263  // this shared resource has been allocated to (and has not been recovered
264  // from) a specific framework.
265  //
266  // NOTE: We keep track of the slave's allocated resources despite
267  // having that information in sorters. This is because the
268  // information in sorters is not accurate if some framework
269  // hasn't reregistered. See MESOS-2919 for details.
270  Resources allocated;
271 
272  // We track the total and allocated resources on the slave to
273  // avoid calculating it in place every time.
274  //
275  // Note that `available` always contains all the shared resources on the
276  // agent regardless whether they have ever been allocated or not.
277  // NOTE, however, we currently only offer a shared resource only if it has
278  // not been offered in an allocation cycle to a framework. We do this mainly
279  // to preserve the normal offer behavior. This may change in the future
280  // depending on use cases.
281  //
282  // Note that it's possible for the slave to be over-allocated!
283  // In this case, allocated > total.
284  Resources available;
285 
286  // We keep a copy of the shared resources to avoid unnecessary copying.
287  Resources shared;
288 
289  // We cache whether the agent has gpus as an optimization.
290  bool hasGpu_;
291 };

559  // We track information about roles that we're aware of in the system.
560  // Specifically, we keep track of the roles when a framework subscribes to
561  // the role, and/or when there are resources allocated to the role

569  // There are two stages of allocation:
570  //
571  // Stage 1: Allocate to satisfy quota guarantees.
572  //
573  // Stage 2: Allocate above quota guarantees up to quota limits.
574  // Note that we need to hold back enough "headroom"
575  // to ensure that any unsatisfied quota can be
576  // satisfied later.
577  //
578  // Each stage comprises two levels of sorting, hence "hierarchical".
579  // Level 1 sorts across roles:
580  // Currently, only the allocated portion of the reserved resources are
581  // accounted for fairness calculation.
582  //
583  // TODO(mpark): Reserved resources should be accounted for fairness
584  // calculation whether they are allocated or not, since they model a long or
585  // forever running task. That is, the effect of reserving resources is
586  // equivalent to launching a task in that the resources that make up the
587  // reservation are not available to other roles as non-revocable.
588  //
589  // Level 2 sorts across frameworks within a particular role:
590  // Reserved resources at this level are, and should be accounted for
591  // fairness calculation only if they are allocated. This is because
592  // reserved resources are fairly shared across the frameworks in the role.
593  //
594  // The allocator relies on `Sorter`s to employ a particular sorting
595  // algorithm. Each level has its own sorter and hence may have different
596  // fairness calculations.
597  //
598  // NOTE: The hierarchical allocator considers revocable resources as
599  // regular resources when doing fairness calculations.
600  //
601  // TODO(vinod): Consider using a different fairness algorithm for
602  // revocable resources.
603 
604  // A sorter for active roles. This sorter determines the order in which
605  // roles are allocated resources during Level 1 of the second stage.
606  // The total cluster resources are used as the resource pool.
607  process::Owned<Sorter> roleSorter;
608 
609  // A collection of sorters, one per active role. Each sorter determines
610  // the order in which frameworks that belong to the same role are allocated
611  // resources inside the role's share. These sorters are used during Level 2
612  // for both the first and the second stages. Since frameworks are sharing
613  // the resources allocated to a role, the role's allocation is used as
614  // the resource pool for each role specific framework sorter.
615  hashmap<std::string, process::Owned<Sorter>> frameworkSorters;
616

618  const std::function<Sorter*()> frameworkSorterFactory;
