33 // An efficient collection of resource quantities. All values are guaranteed
34 // to be positive and finite.
35 //
36 // E.g. [("cpus", 4.5), ("gpus", 0.1), ("ports", 1000)]
37 //
38 // Absent resource entries imply there is no (zero) such resources.
39 //
40 // Notes on handling negativity and arithmetic operations: values are guaranteed
41 // to be positive. This is achieved by construction validation and no public
42 // mutation interfaces. During construction, `Value::Scalar` arguments must be
43 // non-negative and finite, and entries with zero quantities will be dropped
44 // silently. Invalid arguments will result in an error where `Try`
45 // is returned. For arithmetic operations, non-positive values are silently
46 // dropped--this is consist with `class Resources`.
47 //
48 // Note for posterity, the status quo prior to this class
49 // was to use stripped-down `Resources` objects for storing
50 // quantities, however this approach:
51 //
52 // (1) did not support quantities of non-scalar resources;
53 // (2) was error prone, the caller must ensure that no
54 // `Resource` metatdata (e.g. `DiskInfo`) is present;
55 // (3) had poor performance, given the `Resources` storage
56 // model and arithmetic implementation have to operate
57 // on broader invariants.
58 class ResourceQuantities
59 {
60 public:
61  // Parse an input string of semicolon separated "name:number" pairs.
62  // Duplicate names are allowed in the input and will be merged into one entry.
63  // Entries with zero values will be silently dropped.
64  //
65  // Example: "cpus:10;mem:1024;ports:3"
66  // "cpus:10; mem:1024; ports:3"
67  // NOTE: we will trim the whitespace around the pair and in the number.
68  // However, whitespace in "c p us:10" are preserved and will be parsed to
69  // {"c p us", 10}. This is consistent with `Resources::fromSimpleString()`.
70  //
71  // Numbers must be non-negative and finite, otherwise an `Error`
72  // will be returned.
73  static Try<ResourceQuantities> fromString(const std::string& text);
74 
75  // Take scalar `Resources` and combine them into `ResourceQuantities`.
76  // Only the resource name and its scalar value are used and the rest of the
77  // meta-data is ignored. It is caller's responsibility to ensure all
78  // `Resource` entries are of scalar type. Otherwise a `CHECK` error will
79  // be triggered.
80  static ResourceQuantities fromScalarResources(const Resources& resources);
81 
82  // Take `Resources` and combine them into `ResourceQuantities`. This function
83  // assumes that the provided resources have already been validated; for
84  // example, it assumes that ranges do not overlap and that sets do not contain
85  // duplicate items.
86  static ResourceQuantities fromResources(const Resources& resources);
87 
88  // Returns the summed up `ResourceQuantities` given a
89  // `hashmap<Key, ResourceQuantities>`.
90  template <typename Key>
91  static ResourceQuantities sum(const hashmap<Key, ResourceQuantities>& map)
92  {
93  ResourceQuantities result;
94 
95  foreachvalue (const ResourceQuantities& quantities, map) {
96  result += quantities;
97  }
98 
99  return result;
100  }

159 // An efficient collection of resource limits. All values are guaranteed
160 // to be non-negative and finite.
161 //
162 // E.g. [("cpus", 4.5), ("gpus", 0.1), ("ports", 1000)]
163 //
164 // Difference with `class ResourceQuantities`:
165 // The main difference resides in the semantics of absent entries and, in turn,
166 // whether zero values are preserved.
167 //
168 // (1) Absent entry in `ResourceQuantities` implies zero quantity. While in
169 // `ResourceLimits`, absence implies no limit (i.e. infinite amount). This
170 // affects the semantics of the contains operation.
171 //
172 // (2) Zero value preservation: `ResourceLimits` keeps zero value entries
173 // while `ResourceQuantities` silently drops them. This is in accordance with
174 // the absence semantic. Zero value and absence are equivalent in
175 // `ResourceQuantities`. While in `ResourceLimits`, they are not.
176 class ResourceLimits
177 {
178 public:
179  // Parse an input string of semicolon separated "name:number" pairs.
180  // Entries with zero values will be preserved (unlike `ResourceQuantities`).
181  // Duplicate names are NOT allowed in the input, otherwise an `Error` will
182  // be returned.
183  //
184  // Example: "cpus:10;mem:1024;ports:0"
185  // "cpus:10; mem:1024; ports:0"
186  // NOTE: we will trim the whitespace around the pair and in the number.
187  // However, whitespace in "c p us:10" are preserved and will be parsed to
188  // {"c p us", 10}. This is consistent with `Resources::fromSimpleString()`.
189  //
190  // Numbers must be non-negative and finite, otherwise an `Error`
191  // will be returned.
192  static Try<ResourceLimits> fromString(const std::string& text);

221  // Returns the limit of the resource with the given name.
222  // If there is no explicit limit for the resource, return `None()`.
223  // Note, `None()` implies that the limit of the resource is infinite.
224  Option<Value::Scalar> get(const std::string& name) const;

226  // Due to the absence-means-infinite semantic of limits, absent entries,
227  // intuitively, are considered to contain entries with finite scalar values.
228  // For example:
229  // `[ ]`` will always contain any other `ResourceLimits`.
230  // `[("cpu":1)]` contains `[("mem":1)]` and vice versa.
231  // `[("cpu":1)]` will not contain `[("cpu":2)]`.
