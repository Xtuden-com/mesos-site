28 // Converts a fully formed URI to a filename for the platform.
29 //
30 // On all platforms, the optional "file://" prefix is removed if it
31 // exists.
32 //
33 // On Windows, this also converts "/" characters to "\" characters.
34 // The Windows file system APIs don't work with "/" in the filename
35 // when using long paths (although they do work fine if the file
36 // path happens to be short).
37 //
38 // NOTE: Currently, Mesos uses URIs and files somewhat interchangably.
39 // For compatibility, lack of "file://" prefix is not considered an
40 // error.
41 inline std::string from_uri(const std::string& uri)
42 {
43  // Remove the optional "file://" if it exists.
44  // TODO(coffler): Remove the `hostname` component.
45  const std::string path = strings::remove(uri, "file://", strings::PREFIX);
46 
47 #ifndef __WINDOWS__
48  return path;
49 #else
50  return strings::replace(path, "/", "\\");
51 #endif // __WINDOWS__
52 }

55 // Normalizes a given pathname and removes redundant separators and up-level
56 // references.
57 //
58 // Pathnames like `A/B/`, `A///B`, `A/./B`, 'A/foobar/../B` are all normalized
59 // to `A/B`. An empty pathname is normalized to `.`. Up-level entry also cannot
60 // escape a root path, in which case an error will be returned.
61 //
62 // This function follows the rules described in path_resolution(7) for Linux.
63 // However, it only performs pure lexical processing without touching the
64 // actual filesystem.
65 inline Try<std::string> normalize(
66  const std::string& path,
67  const char _separator = os::PATH_SEPARATOR)
68 {
69  if (path.empty()) {
70  return ".";
71  }
72 
73  std::vector<std::string> components;
74  const bool isAbs = (path[0] == _separator);
75  const std::string separator(1, _separator);
76 
77  // TODO(jasonlai): Handle pathnames (including absolute paths) in Windows.
78 
79  foreach (const std::string& component, strings::tokenize(path, separator)) {
80  // Skips empty components and "." (current directory).
81  if (component == "." || component.empty()) {
82  continue;
83  }
84 
85  if (component == "..") {
86  if (components.empty()) {
87  if (isAbs) {
88  return Error("Absolute path '" + path + "' tries to escape root");
89  }
90  components.push_back(component);
91  } else if (components.back() == "..") {
92  components.push_back(component);
93  } else {
94  components.pop_back();
95  }
96  } else {
97  components.push_back(component);
98  }
99  }
100 
101  if (components.empty()) {
102  return isAbs ? separator : ".";
103  } else if (isAbs) {
104  // Make sure that a separator is prepended if it is an absolute path.
105  components.insert(components.begin(), "");
106  }
107 
108  return strings::join(separator, components);
109 }

153 inline bool absolute(const std::string& path)
154 {
155 #ifndef __WINDOWS__
156  return strings::startsWith(path, os::PATH_SEPARATOR);
157 #else
158  // NOTE: We do not use `PathIsRelative` Windows utility function
159  // here because it does not support long paths.
160  //
161  // See https://msdn.microsoft.com/en-us/library/windows/desktop/aa365247(v=vs.85).aspx
162  // for details on paths. In short, an absolute path for files on Windows
163  // looks like one of the following:
164  // * "[A-Za-z]:\"
165  // * "[A-Za-z]:/"
166  // * "\\?\..."
167  // * "\\server\..." where "server" is a network host.
168  //
169  // NOLINT(whitespace/line_length)
170 
171  // A uniform naming convention (UNC) name of any format,
172  // always starts with two backslash characters.
173  if (strings::startsWith(path, "\\\\")) {
174  return true;
175  }
176 
177  // A disk designator with a slash, for example "C:\" or "d:/".
178  if (path.length() < 3) {
179  return false;
180  }
181 
182  const char letter = path[0];
183  if (!((letter >= 'A' && letter <= 'Z') ||
184  (letter >= 'a' && letter <= 'z'))) {
185  return false;
186  }
187 
188  std::string colon = path.substr(1, 2);
189  return colon == ":\\" || colon == ":/";
190 #endif // __WINDOWS__
191 }

239  inline std::string basename() const
240  {
241  if (value.empty()) {
242  return std::string(".");
243  }
244 
245  size_t end = value.size() - 1;
246 
247  // Remove trailing slashes.
248  if (value[end] == separator) {
249  end = value.find_last_not_of(separator, end);
250 
251  // Paths containing only slashes result into "/".
252  if (end == std::string::npos) {
253  return stringify(separator);
254  }
255  }
256 
257  // 'start' should point towards the character after the last slash
258  // that is non trailing.
259  size_t start = value.find_last_of(separator, end);
260 
261  if (start == std::string::npos) {
262  start = 0;
263  } else {
264  start++;
265  }
266 
267  return value.substr(start, end + 1 - start);
268  }

305 
306  // Remove trailing slashes.
307  if (value[end] == separator) {
308  end = value.find_last_not_of(separator, end);
309  }
310 
311  // Remove anything trailing the last slash.
312  end = value.find_last_of(separator, end);
313 
314  // Paths containing no slashes result in ".".
315  if (end == std::string::npos) {
316  return std::string(".");
317  }
318 
319  // Paths containing only slashes result in "/".
320  if (end == 0) {
321  return stringify(separator);
322  }
323 
324  // 'end' should point towards the last non slash character
325  // preceding the last slash.
326  end = value.find_last_not_of(separator, end);
327 
328  // Paths containing no non slash characters result in "/".
329  if (end == std::string::npos) {
330  return stringify(separator);
331  }
332

334  }
