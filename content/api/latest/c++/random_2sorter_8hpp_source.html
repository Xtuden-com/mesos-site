204 // Represents a node in the sorter's tree. The structure of the tree
205 // reflects the hierarchical relationships between the clients of the
206 // sorter. Some (but not all) nodes correspond to sorter clients; some
207 // nodes only exist to represent the structure of the sorter
208 // tree. Clients are always associated with leaf nodes.
209 //
210 // For example, if there are two sorter clients "a/b" and "c/d", the
211 // tree will contain five nodes: the root node, internal nodes for "a"
212 // and "c", and leaf nodes for the clients "a/b" and "c/d".
213 struct RandomSorter::Node
214 {
215  // Indicates whether a node is an active leaf node, an inactive leaf
216  // node, or an internal node. Sorter clients always correspond to
217  // leaf nodes, and only leaf nodes can be activated or deactivated.
218  // The root node is always an "internal" node.
219  enum Kind
220  {
221  ACTIVE_LEAF,
222  INACTIVE_LEAF,
223  INTERNAL
224  };
225 
226  Node(const std::string& _name, Kind _kind, Node* _parent)
227  : name(_name), kind(_kind), parent(_parent)
228  {
229  // Compute the node's path. Three cases:
230  //
231  // (1) If the root node, use the empty string
232  // (2) If a child of the root node, use the child's name
233  // (3) Otherwise, use the parent's name, "/", and the child's name.
234  if (parent == nullptr) {
235  path = "";
236  } else if (parent->parent == nullptr) {
237  path = name;
238  } else {
239  path = strings::join("/", parent->path, name);
240  }
241  }
242 
243  ~Node()
244  {
245  foreach (Node* child, children) {
246  delete child;
247  }
248  }
249 
250  // The label of the edge from this node's parent to the
251  // node. "Implicit" leaf nodes are always named ".".
252  //
253  // TODO(neilc): Consider naming implicit leaf nodes in a clearer
254  // way, e.g., by making `name` an Option?
255  std::string name;
256 
257  // Complete path from root to node. This includes the trailing "."
258  // label for virtual leaf nodes.
259  std::string path;
260 
261  // Cached weight of the node, access this through `getWeight()`.
262  // The value is cached by `getWeight()` and updated by
263  // `updateWeight()`. Marked mutable since the caching writes
264  // to this are logically const.
265  mutable Option<double> weight;
266 
267  Kind kind;
268 
269  Node* parent;
270 
271  // Pointers to the child nodes. `children` is only non-empty if
272  // `kind` is INTERNAL_NODE.
273  //
274  // All inactive leaves are stored at the end of the vector; that
275  // is, each `children` vector consists of zero or more active leaves
276  // and internal nodes, followed by zero or more inactive leaves. This
277  // means that code that only wants to iterate over active children
278  // can stop when the first inactive leaf is observed.
279  std::vector<Node*> children;
280 
281  // If this node represents a sorter client, this returns the path of
282  // that client. Unlike the `path` field, this does NOT include the
283  // trailing "." label for virtual leaf nodes.
284  //
285  // For example, if the sorter contains two clients "a" and "a/b",
286  // the tree will contain four nodes: the root node, "a", "a/."
287  // (virtual leaf), and "a/b". The `clientPath()` of "a/." is "a",
288  // because that is the name of the client associated with that
289  // virtual leaf node.
290  const std::string& clientPath() const
291  {
292  if (name == ".") {
293  CHECK(kind == ACTIVE_LEAF || kind == INACTIVE_LEAF);
294  return CHECK_NOTNULL(parent)->path;
295  }
296 
297  return path;
298  }
299 
300  bool isLeaf() const
301  {
302  if (kind == ACTIVE_LEAF || kind == INACTIVE_LEAF) {
303  CHECK(children.empty());
304  return true;
305  }
306 
307  return false;
308  }

310  void removeChild(const Node* child)
311  {
312  // Sanity check: ensure we are removing an extant node.
313  auto it = std::find(children.begin(), children.end(), child);
314  CHECK(it != children.end());
315 
316  children.erase(it);
317  }
318 
319  void addChild(Node* child)
320  {
321  // Sanity check: don't allow duplicates to be inserted.
322  auto it = std::find(children.begin(), children.end(), child);
323  CHECK(it == children.end());
324 
325  // If we're inserting an inactive leaf, place it at the end of the
326  // `children` vector; otherwise, place it at the beginning. This
327  // maintains ordering invariant above.
328  if (child->kind == INACTIVE_LEAF) {
329  children.push_back(child);
330  } else {
331  children.insert(children.begin(), child);
332  }
333  }

340  void add(const SlaveID& slaveId, const Resources& toAdd)
341  {
342  // Add shared resources to the allocated quantities when the same
343  // resources don't already exist in the allocation.
344  const Resources sharedToAdd = toAdd.shared()
345  .filter([this, slaveId](const Resource& resource) {
346  return !resources[slaveId].contains(resource);
347  });
348 
349  const ResourceQuantities quantitiesToAdd =
350  ResourceQuantities::fromScalarResources(
351  (toAdd.nonShared() + sharedToAdd).scalars());
352 
353  resources[slaveId] += toAdd;
354  totals += quantitiesToAdd;
355  }
356 
357  void subtract(const SlaveID& slaveId, const Resources& toRemove)
358  {
359  CHECK(resources.contains(slaveId));
360  CHECK(resources.at(slaveId).contains(toRemove))
361  << "Resources " << resources.at(slaveId) << " at agent " << slaveId
362  << " does not contain " << toRemove;
363 
364  resources[slaveId] -= toRemove;
365 
366  // Remove shared resources from the allocated quantities when there
367  // are no instances of same resources left in the allocation.
368  const Resources sharedToRemove = toRemove.shared()
369  .filter([this, slaveId](const Resource& resource) {
370  return !resources[slaveId].contains(resource);
371  });
372 
373  const ResourceQuantities quantitiesToRemove =
374  ResourceQuantities::fromScalarResources(
375  (toRemove.nonShared() + sharedToRemove).scalars());
376 
377  CHECK(totals.contains(quantitiesToRemove))
378  << totals << " does not contain " << quantitiesToRemove;
379 
380  totals -= quantitiesToRemove;
381 
382  if (resources[slaveId].empty()) {
383  resources.erase(slaveId);
384  }
385  }

387  void update(
388  const SlaveID& slaveId,
389  const Resources& oldAllocation,
390  const Resources& newAllocation)
391  {
392  const ResourceQuantities oldAllocationQuantities =
393  ResourceQuantities::fromScalarResources(oldAllocation.scalars());
394  const ResourceQuantities newAllocationQuantities =
395  ResourceQuantities::fromScalarResources(newAllocation.scalars());
396 
397  CHECK(resources.contains(slaveId));
398  CHECK(resources[slaveId].contains(oldAllocation))
399  << "Resources " << resources[slaveId] << " at agent " << slaveId
400  << " does not contain " << oldAllocation;
401 
402  CHECK(totals.contains(oldAllocationQuantities))
403  << totals << " does not contain " << oldAllocationQuantities;
404 
405  resources[slaveId] -= oldAllocation;
406  resources[slaveId] += newAllocation;
407 
408  totals -= oldAllocationQuantities;
409  totals += newAllocationQuantities;
410  }
