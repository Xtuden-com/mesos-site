31 // The file system layout is as follows:
32 //
33 // root (<work_dir>/csi/)
34 // |-- <type>
35 // |-- <name>
36 // |-- containers
37 // | |-- <container_id>
38 // | |-- container.info
39 // | |-- endpoint (symlink to /tmp/mesos-csi-XXXXXX)
40 // | |-- endpoint.sock
41 // |-- volumes
42 // | |-- <volume_id>
43 // | |-- volume.state
44 // |-- mounts
45 // |-- <volume_id>
