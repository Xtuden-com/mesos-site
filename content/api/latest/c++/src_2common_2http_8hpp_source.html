106 // Deserializes a string message into a protobuf message based on the
107 // HTTP content type.
108 template <typename Message>
109 Try<Message> deserialize(
110  ContentType contentType,
111  const std::string& body)
112 {
113  switch (contentType) {
114  case ContentType::PROTOBUF: {
115  Message message;
116  if (!message.ParseFromString(body)) {
117  return Error("Failed to parse body into a protobuf object");
118  }
119  return message;
120  }
121  case ContentType::JSON: {
122  Try<JSON::Value> value = JSON::parse(body);
123  if (value.isError()) {
124  return Error("Failed to parse body into JSON: " + value.error());
125  }
126 
127  return ::protobuf::parse<Message>(value.get());
128  }
129  case ContentType::RECORDIO: {
130  return Error("Deserializing a RecordIO stream is not supported");
131  }
132  }
133 
134  UNREACHABLE();
135 }

241  template <authorization::Action action, typename... Args>
242  bool approved(const Args&... args)
243  {
244  if (!approvers.contains(action)) {
245  LOG(WARNING) << "Attempted to authorize " << principal
246  << " for unexpected action " << stringify(action);
247  return false;
248  }
249 
250  Try<bool> approved = approvers[action]->approved(
251  ObjectApprover::Object(args...));
252 
253  if (approved.isError()) {
254  // TODO(joerg84): Expose these errors back to the caller.
255  LOG(WARNING) << "Failed to authorize principal " << principal
256  << "for action " << stringify(action) << ": "
257  << approved.error();
258  return false;
259  }
260 
261  return approved.get();
262  }

270  : approvers(std::move(_approvers)),

281 template <>
282 inline bool ObjectApprovers::approved<authorization::VIEW_ROLE>(
283  const Resource& resource)
284 {
285  // Necessary because recovered agents are presented in old format.
286  if (resource.has_role() && resource.role() != "*" &&
287  !approved<authorization::VIEW_ROLE>(resource.role())) {
288  return false;
289  }
290 
291  // Reservations follow a path model where each entry is a child of the
292  // previous one. Therefore, to accept the resource the acceptor has to
293  // accept all entries.
294  foreach (Resource::ReservationInfo reservation, resource.reservations()) {
295  if (!approved<authorization::VIEW_ROLE>(reservation.role())) {
296  return false;
297  }
298  }
299 
300  if (resource.has_allocation_info() &&
301  !approved<authorization::VIEW_ROLE>(
302  resource.allocation_info().role())) {
303  return false;
304  }
305 
306  return true;
307 }
